📌 О проекте FileHub

FileHub — это клиент-серверная система для обмена файлами в реальном времени. Сервер координирует подключения пользователей, хранит метаданные раздач и проксирует передачу данных между отправителем и получателем. Клиент предоставляет консольный интерфейс для взаимодействия с сервером.

Ключевые особенности архитектуры:

  • 🔹 Модульная структура с внедрением зависимостей
  • 🔹 Единый протокол обмена на основе JSON
  • 🔹 Разделение управляющего канала и канала передачи данных
  • 🔹 Конфигурация через .conf файлы
  • 🔹 Расширяемость через добавление новых модулей

🏗️ Общая архитектура системы

📡 Протокол обмена

Файл: comments.py

"""
Модуль комментариев к раздачам
"""

import time
from datetime import datetime

class CommentManager:
    """Менеджер комментариев к раздачам"""
    
    def __init__(self, storage, logger):
        self.storage = storage
        self.logger = logger
        
        # Структура: {share_id: [{username, text, timestamp}]}
        self.comments = {}
        
        self.logger.log('INFO', 'CommentManager инициализирован')
    
    def add_comment(self, share_id, username, text):
        """
        Добавление комментария к раздаче
        
        Аргументы:
            share_id: ID раздачи
            username: имя пользователя
            text: текст комментария
        """
        if share_id not in self.storage.shares:
            return False, 'Раздача не найдена'
        
        if not text or len(text) > 500:
            return False, 'Комментарий должен быть от 1 до 500 символов'
        
        if share_id not in self.comments:
            self.comments[share_id] = []
        
        comment = {
            'username': username,
            'text': text,
            'timestamp': datetime.now().isoformat()
        }
        
        self.comments[share_id].append(comment)
        
        # Ограничиваем количество комментариев
        if len(self.comments[share_id]) > 100:
            self.comments[share_id] = self.comments[share_id][-100:]
        
        self.logger.log('INFO', f'Комментарий от {username} к раздаче {share_id}')
        return True, 'Комментарий добавлен'
    
    def get_comments(self, share_id):
        """
        Получение комментариев к раздаче
        
        Аргументы:
            share_id: ID раздачи
        """
        return self.comments.get(share_id, [])

        Инициализация модуля
        
        Аргументы:
            connection: ConnectionManager
            config: ClientConfig
        """
        self.connection = connection
        self.config = config
        self.data = []
    
    def do_something(self, param):

        else:
            error = response.get('message', 'Ошибка') if response else 'Нет ответа'

            return False

📋 Полный пример: модуль поиска по раздачам

Файл: search.py

"""
Модуль поиска по раздачам
"""

from protocols import MessageActions
from console_ui import console

class SearchManager:
    """Менеджер поиска раздач"""
    
    def __init__(self, connection):
        self.connection = connection
    
    def search(self, query):
        """
        Поиск раздач по имени или владельцу
        
        Аргументы:
            query: поисковый запрос
        """
        if not query:
            console.print_system_message('Введите поисковый запрос', 'warning')
            return
        
        # Получаем все раздачи
        response = self.connection.send_command({
            'action': MessageActions.LIST
        })
        
        if not response or response.get('status') != 'ok':
            console.print_system_message('Не удалось получить список раздач', 'error')
            return
        
        shares = response.get('shares', [])
        query_lower = query.lower()
        
        # Фильтруем локально
        results = [
            s for s in shares
            if query_lower in s['name'].lower() or 
               query_lower in s['username'].lower()
        ]
        
        if not results:
            console.print_system_message(f'По запросу "{query}" ничего не найдено', 'info')
            return
        
        print(f'\nРезультаты поиска "{query}" (найдено: {len(results)}):')
        print('=' * 70)
        
        for share in results:
            icon = '📁' if share['type'] == 'folder' else '📄'
            status = '🟢' if share.get('owner_online') else '🔴'
            print(f'{icon} [{share["share_id"]}] {share["name"]}')
            print(f'   {status} Владелец: {share["username"]} | Скачиваний: {share.get("downloads", 0)}')
            print('-' * 70)
    
    def search_by_extension(self, extension):
        """
        Поиск файлов по расширению
        
        Аргументы:
            extension: расширение файла (например, '.pdf')
        """
        if not extension.startswith('.'):
            extension = '.' + extension
        
        response = self.connection.send_command({
            'action': MessageActions.LIST
        })
        
        if not response or response.get('status') != 'ok':
            console.print_system_message('Не удалось получить список раздач', 'error')
            return
        
        shares = response.get('shares', [])
        
        results = [
            s for s in shares
            if s['type'] == 'file' and s['name'].lower().endswith(extension.lower())
        ]
        
        if not results:
            console.print_system_message(f'Файлы {extension} не найдены', 'info')
            return
        
        print(f'\nФайлы {extension} (найдено: {len(results)}):')
        print('=' * 70)
        for share in results:
            print(f'📄 [{share["share_id"]}] {share["name"]}')
            print(f'   Владелец: {share["username"]} | Размер: {share.get("size", 0)} байт')
        print('=' * 70)

✅ Лучшие практики при добавлении модулей

Принципы проектирования

  • Единая ответственность — каждый модуль решает одну задачу
  • Внедрение зависимостей — зависимости передаются через конструктор
  • Не изменяйте ядро — добавляйте новые действия в протокол, не модифицируя существующие
  • Логирование — используйте logger.log() для важных событий
  • Обработка ошибок — всегда возвращайте понятные сообщения об ошибках

Что следует избегать

  • ❌ Прямого доступа к внутренним данным других модулей
  • ❌ Блокирующих операций в основном потоке
  • ❌ Жестких зависимостей от конкретных реализаций
  • ❌ Изменения формата существующих сообщений протокола
